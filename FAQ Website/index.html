Write a function that will reverse the given input string.

string SwapLetters (string word) {

string swappedword = "";


for (int i = word.length() - 1; i >= 0; --i) {

swappedword.push_back(word.at(i));

}


return swappedword;

}

Write a function that accepts three integers as input and returns as output the largest of the three.

int CalculateMax (int num1, int num2, int num3) {

int maxInt = num1;

if (maxInt < num2) { maxInt = num2; }

if (maxInt < num3) { maxInt = num3; }


return maxInt;

}

Write a function that calculates the factorial of an input integer using recursion.

int Factorial(int num) {

if (num > 1) { return num * Factorial(num - 1); }

else { return 1; }

}

Write a function that calculates the Nth entry of the Fibonaccisequence.

int CalcFibonacci (int n) {

int fibSeq[n];

fibSeq[0] = 1;

fibSeq[1] = 1;


for (int i = 2; i < n; ++i) {

fibSeq[i] = fibSeq[i - 1] + fibSeq[i - 2];

}


int answer = fibSeq[n-1];


return answer;

}
